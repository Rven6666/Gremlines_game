package gremlins;

import processing.core.PImage;
import processing.core.PApplet;

public class Brickwalls extends Walls {

    private PImage[] imageState;
    private int counter;
    private int imageNumber;
    protected Boolean hit;

    public Brickwalls(App app, int x, int y, PImage[] imageState, boolean solidTile) {
        super(app, x, y);
        this.imageState = imageState;
        this.hit = false;
        this.counter = 1;
        this.imageNumber = 0;
        this.destroyed = false;
        this.solidTile = solidTile;
        this.app = app;
    }

    public void tick() {

        if (hit == true) { // Brick Destroyed when hit
            if (counter == 1) {
                imageNumber = checkDestruction(imageNumber);
            } else if (counter == 4) {
                imageNumber = checkDestruction(imageNumber);
            } else if (counter == 8) {
                imageNumber = checkDestruction(imageNumber);
            } else if (counter == 12) {
                imageNumber = checkDestruction(imageNumber);
            } else if (counter == 16) {
                imageNumber = checkDestruction(imageNumber);
            } else if (counter >= 20) {
                this.destroyed = true;
            }
            counter++;
        }
    }

    public void draw(PApplet app) {
        if (destroyed == false) {
            if (hit == false) { // normal brick state
                app.image(this.imageState[0], this.x, this.y);
            } else if (counter < 20 && imageNumber < 5) {
                app.image(this.imageState[imageNumber], this.x, this.y);
            }
        }
    }

    // increments the images of brick to be destroyed when hit
    public int checkDestruction(int currentImage) {
        return ++currentImage;
    }

    // starts animation sequence if hit by fireball
    public void fireCollision() {
        this.hit = true;
    }

    // checks if brickwall has been destroyed
    public boolean getDestroyed() {
        return this.destroyed;
    }

}
