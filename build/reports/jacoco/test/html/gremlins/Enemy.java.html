package gremlins;

import java.util.Random;

import processing.core.PApplet;
import processing.core.PImage;
import java.util.ArrayList;

public class Enemy extends Characters {

    protected PImage sprite;
    protected int low;
    protected int high;
    protected int backTrack;
    protected boolean respawn;
    protected Boolean freeze;
    protected Map respawnCoord;
    protected Random newDirec;
    protected ArrayList<Integer> posDirec;

    public Enemy(App app, int x, int y, double coolDown, PImage sprite) {
        super(app, x, y, coolDown);
        this.sprite = sprite;
        this.low = 0;
        this.high = 4;
        this.backTrack = 4;
        this.freeze = false;
        this.respawnCoord = new Map(app);
        this.speed = 1;
        this.direction = App.randomGenerator.nextInt(4);
        this.app = app;
        this.respawn = false;
        this.moveControl = false;
        this.moving = true;
        this.coolDownFrames = (int) Math.round(App.FPS * this.coolDown);
        this.newDirec = new Random();

    }

    public void tick() {
        super.tick();

        if (isAlive == true) {

            if (this.app.colCheck.checkFireballCollision(this.x, this.y, this.spriteSize) == true) {
                this.isAlive = false;
            }

            // checks if collision is true, if so makes sure direction is unique
            if (this.collision == true) {
                // randomises enemy direction on collision
                if (direction == 1) {
                    int[] ran = new int[] { 2, 3 };
                    int rnd = new Random().nextInt(ran.length);
                    this.direction = ran[rnd];
                } else if (direction == 0) {
                    int[] ran = new int[] { 2, 3 };
                    int rnd = new Random().nextInt(ran.length);
                    this.direction = ran[rnd];
                } else if (direction == 2) {
                    int[] ran = new int[] { 1, 0 };
                    int rnd = new Random().nextInt(ran.length);
                    this.direction = ran[rnd];
                } else if (direction == 3) {
                    int[] ran = new int[] { 1, 0 };
                    int rnd = new Random().nextInt(ran.length);
                    this.direction = ran[rnd];
                }
            }
        } else if (respawn == false) {
            // respawns enemies to random location 10+ spaces away rom wizard.
            int[] randCoord = new int[1];
            randCoord = respawnCoord.mapRandomSpawn();
            this.x = randCoord[0];
            this.y = randCoord[1];
            this.isAlive = true;
            respawn = false;
        }

    }

    public void draw(PApplet app) {
    }
}
