package gremlins;

import processing.core.PApplet;
import processing.core.PImage;

public class ExitPortal {
    private int x;
    private int y;
    private PImage sprite;

    public ExitPortal(int x, int y, PImage sprite) {
        this.x = x;
        this.y = y;
        this.sprite = sprite;
    }

    void draw(PApplet app) {
        app.image(this.sprite, this.x, this.y);
    }

    // Get X Value
    public int getX() {
        return this.x;
    }

    // Get Y Value
    public int getY() {
        return this.y;
    }

}
