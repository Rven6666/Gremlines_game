package gremlins;

import processing.core.PImage;

public class Fireball extends Projectile {

    public Fireball(App app, int x, int y, boolean shotFired, int direction, PImage sprite) {
        super(app, x, y, shotFired, direction, sprite);
        this.type = "fire";
    }

    @Override
    public void tick() {

        super.tick();
        if (collision == false) {
            if (this.app.colCheck.checkSlimeballCollision(this.x, this.y,
                    this.spriteSize) == true) {
                this.collision = true;
            }
            if (this.app.colCheck.checkPumkinCollision(this.x, this.y,
                    this.spriteSize) == true) {
                this.collision = true;
            }
        }
        if (this.app.colCheck.checkBrickwallCollision(this.x, this.y,
                this.spriteSize, this.type) == true) {
            this.collision = true;
        }
        if (this.collision == true) {
            this.app.fireballList.remove(this.toString());
        }
    }
}
