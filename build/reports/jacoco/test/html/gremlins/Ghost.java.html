package gremlins;

import processing.core.PApplet;
import processing.core.PImage;

public class Ghost extends Enemy {

    private Timer direcTime;
    private int randomDirec;

    public Ghost(App app, double ghostCoolDown, int x, int y, PImage ghostSprite) {
        super(app, x, y, ghostCoolDown, ghostSprite);
        this.speed = 1;
        this.shotReady = true;
        this.direcTime = new Timer(3);
        this.randomDirec = App.randomGenerator.nextInt(4);
        ;

    }

    @Override
    public void tick() {
        super.tick();
        if (isAlive) {
            if (this.shotReady == true) {
                this.app.pumkinballList.add(new Pumkin(app, this.x, this.y, true, this.direction, app.pumkinSprite));
                this.shotReady = false;
            }
        }

        direcTime.countDown();
        if (direcTime.getTime() <= 0) {
            this.randomDirec = App.randomGenerator.nextInt(4);

            if (collision == false && this.x % 20 == 0 && this.y % 20 == 0) {
                this.direction = randomDirec;
                this.direcTime.setTime(App.randomGenerator.nextInt(6));
            }
        }
    }

    @Override
    public void draw(PApplet app) {
        super.draw(app);
        if (isAlive) {
            app.image(this.sprite, this.x, this.y);
        }
    }
}
