package gremlins;

import processing.core.PApplet;
import processing.core.PImage;

public class Gremlins extends Enemy {

    public Gremlins(App app, double enemyCoolDown, int x, int y, PImage sprite) {
        super(app, x, y, enemyCoolDown, sprite);
        this.shotReady = true;
    }

    public void tick() {
        super.tick();

        if (isAlive == true) {
            if (this.shotReady == true && freeze == false) {
                this.app.slimeballList.add(new Slimeball(app, this.x, this.y, true, this.direction, app.slimeSprite));
                this.shotReady = false;
            }
        }
    }

    public void draw(PApplet app) {
        if (getIsAlive() == true) {
            if (this.freeze == true) {
                app.image(this.app.getFrozenGrem(), this.x, this.y);
            } else {
                app.image(this.sprite, this.x, this.y);
            }

        }
    }

    // freezes gremlins when power up is picked up
    public void setFreeze(boolean freeze) {
        this.freeze = freeze;

    }

}
