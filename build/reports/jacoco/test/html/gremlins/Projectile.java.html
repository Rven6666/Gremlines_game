package gremlins;

import processing.core.PApplet;
import processing.core.PImage;

public class Projectile {

    protected App app;
    protected int x;
    protected int y;
    protected PImage sprite;
    protected int speed;
    protected int spriteSize;
    protected boolean shotFired;
    protected int direction;
    protected boolean collision;
    protected String type;

    public Projectile(App app, int x, int y, boolean shotFired, int direction, PImage sprite) {
        this.speed = 4;
        this.app = app;
        this.x = x;
        this.y = y;
        this.sprite = sprite;
        this.shotFired = shotFired;
        this.direction = direction;
        this.collision = false;
        this.spriteSize = App.SPRITESIZE;
    }

    public void tick() {

        if (collision == false) {

            if (this.app.colCheck.checkStoneCollision(this.x, this.y, this.spriteSize) == true) {
                this.collision = true;
            }

            // if statement to check which direction the projectile has been shot.
            if (this.direction == 0) {
                left();
            } else if (this.direction == 1) {
                right();
            } else if (this.direction == 2) {
                up();
            } else if (this.direction == 3) {
                down();
            }
        }
    }

    public void draw(PApplet app) {
        if (collision == false) {
            app.image(this.sprite, this.x, this.y);
        }
    }

    // Gets X Value
    public int getX() {
        return this.x;
    }

    // Gets Y value
    public int getY() {
        return this.y;
    }

    // sets Movement to right direction
    public void right() {
        this.x += this.speed;
    }

    // sets Movement to Left direction
    public void left() {
        this.x -= this.speed;
    }

    // sets Movement to up direction
    public void up() {
        this.y -= this.speed;
    }

    // sets Movement to down direction
    public void down() {
        this.y += this.speed;
    }

    // Sets if there has been a collision
    public void setCollision(boolean collision) {
        this.collision = collision;
    }

    // returns if shot has been fired
    public boolean getShot() {
        return this.shotFired;
    }

    // Sets the value when shoe it fired
    public boolean setShot(boolean shotFired) {
        return this.shotFired;
    }

    // Checks if there has been a collision
    public boolean getCollision() {
        return this.collision;
    }

}
